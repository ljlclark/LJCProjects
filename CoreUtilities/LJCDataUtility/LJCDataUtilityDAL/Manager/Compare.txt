DECLARE @constraintName nvarchar(60);
SET @constraintName = 'pk_DataTable';
SELECT tc.TABLE_CATALOG, tc.TABLE_SCHEMA, tc.TABLE_NAME, CONSTRAINT_TYPE, tc.CONSTRAINT_NAME,
 kcu.COLUMN_NAME, kcu.ORDINAL_POSITION, rc.UNIQUE_CONSTRAINT_NAME
FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS as tc
LEFT JOIN INFORMATION_SCHEMA.KEY_COLUMN_USAGE as kcu
 ON tc.CONSTRAINT_NAME = kcu.CONSTRAINT_NAME
LEFT JOIN INFORMATION_SCHEMA.REFERENTIAL_CONSTRAINTS as rc
 ON tc.CONSTRAINT_NAME = rc.CONSTRAINT_NAME
/*WHERE tc.TABLE_NAME = 'DataColumn'
 and CONSTRAINT_TYPE = 'FOREIGN KEY';*/
WHERE tc.CONSTRAINT_NAME = @constraintName;
