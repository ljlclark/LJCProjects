Add a Province using ProvinceSQLManager

The ProvinceSQLManager class is a strongly typed manager class specifically for use with the Province table and Province Data Object. 

// Add a data record.
internal void AddProvince(string connectionString, string providerName)
{
  var manager = new ProvinceSQLManager(null, null, connectionString
  , providerName);

  // Create the Data Object.
  var province = new Province()
  {
    RegionID = 1,
    Name = "ProvinceName",
    Description = "A Sample Province",
    Abbreviation = "PN"
  };
  var resultProvince = manager.Add(province);
  var addedID = resultProvince.ID;
}
